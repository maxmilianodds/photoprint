export default { async fetch(request, env) { const corsHeaders = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "GET, POST, OPTIONS, DELETE", "Access-Control-Allow-Headers": "Content-Type", }; if (request.method === "OPTIONS") return new Response(null, { headers: corsHeaders }); const url = new URL(request.url); // Ajuste no path para não perder subpastas const path = url.pathname.split('/').filter(Boolean).join('/'); const BUCKET = env.PHOTOPRINT_R2; const DB = env.DB; try { // --- ROTA: STATUS DO EVENTO --- if (path === "status") { const config = await DB.prepare("SELECT * FROM config WHERE id = 1").first(); return new Response(JSON.stringify(config), { headers: { ...corsHeaders, "Content-Type": "application/json" } }); } // --- ROTA: LISTAR EVENTOS --- if (path === "list-events") { const events = await DB.prepare("SELECT * FROM config").all(); return new Response(JSON.stringify(events.results), { headers: { ...corsHeaders, "Content-Type": "application/json" } }); } // --- ROTA: DEFINIR EVENTO ATIVO --- if (path === "set-config" && request.method === "POST") { const data = await request.json(); await DB.prepare("UPDATE config SET evento = ?, status = ?, formato = ? WHERE id = 1") .bind(data.evento, data.status, data.formato) .run(); return new Response(JSON.stringify({ success: true }), { headers: corsHeaders }); } // --- ROTA: UPLOAD --- if (path === "upload" && request.method === "POST") { const formData = await request.formData(); const file = formData.get('file'); const evento = formData.get('evento') || "Geral"; const tipo = formData.get('tipo'); if (!file) return new Response("Sem arquivo", { status: 400, headers: corsHeaders }); const fileName = tipo === 'moldura' ? `${evento}/moldura.png` : `${evento}/foto_${Date.now()}.jpg`; await BUCKET.put(fileName, file, { httpMetadata: { contentType: tipo === 'moldura' ? "image/png" : "image/jpeg" } }); return new Response(JSON.stringify({ success: true, path: fileName }), { headers: corsHeaders }); } // --- ROTA: DELETAR ARQUIVO ESPECÍFICO --- if (path === "delete-file" && request.method === "DELETE") { const { key } = await request.json(); if (!key) return new Response("Key não informada", { status: 400, headers: corsHeaders }); await BUCKET.delete(key); return new Response(JSON.stringify({ success: true }), { headers: corsHeaders }); } // --- ROTA: LISTAR FOTOS --- if (path === "list-photos") { const evento = url.searchParams.get("evento"); if (!evento) return new Response("[]", { headers: corsHeaders }); const list = await BUCKET.list({ prefix: `${evento}/` }); const arquivos = list.objects.map(obj => ({ key: obj.key })); return new Response(JSON.stringify(arquivos), { headers: corsHeaders }); } // --- ROTA: SERVIR FOTO (CORRIGIDA PARA SUBPASTAS) --- if (url.pathname.startsWith("/photo/")) { // Pega tudo que vem depois de "/photo/" inclusive subpastas const key = decodeURIComponent(url.pathname.substring(7)); const object = await BUCKET.get(key); if (!object) return new Response("Foto não encontrada no R2", { status: 404, headers: corsHeaders }); const headers = new Headers(corsHeaders); object.writeHttpMetadata(headers); // Garante que o navegador entenda que é uma imagem const contentType = key.toLowerCase().endsWith(".png") ? "image/png" : "image/jpeg"; headers.set("content-type", contentType); headers.set("etag", object.httpEtag); headers.set("Cache-Control", "public, max-age=31536000"); // Cache de 1 ano para fotos return new Response(object.body, { headers }); } // --- ROTA: DELETAR TUDO (Exceto Molduras) --- if (path === "delete-all" && request.method === "DELETE") { const evento = url.searchParams.get("evento") || ""; const list = await BUCKET.list({ prefix: evento ? `${evento}/` : "" }); for (const obj of list.objects) { if (!obj.key.toLowerCase().endsWith('.png')) await BUCKET.delete(obj.key); } return new Response(JSON.stringify({ success: true }), { headers: corsHeaders }); } return new Response("API PHOTO PRINT ONLINE", { headers: corsHeaders }); } catch (e) { return new Response(JSON.stringify({ error: e.message }), { status: 500, headers: { ...corsHeaders, "Content-Type": "application/json" } }); } } };
